CREATE DATABASE sistema;

USE sistema;

-- Tabela de Formas de Pagamento
CREATE TABLE formas_pagamento (
    id INT AUTO_INCREMENT PRIMARY KEY,
    descricao VARCHAR(50) NOT NULL
);

-- Tabela de Condições de Pagamento
CREATE TABLE condicoes_pagamento (
    id INT AUTO_INCREMENT PRIMARY KEY,
    prazo_dias INT NOT NULL
);

-- Tabela de Parcelamento 
CREATE TABLE parcelamento (
    numero_parcela INT,
    cod_condicao_pag INT,
    cod_forma_pag INT,
    PRIMARY KEY (numero_parcela, cod_condicao_pag),
    FOREIGN KEY (cod_condicao_pag) REFERENCES condicao_pagamento(cod_condicao_pag),
    FOREIGN KEY (cod_forma_pag) REFERENCES forma_pagamento(cod_forma_pag)
);
